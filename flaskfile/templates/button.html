As the same, for creating any widget, you must write its name(in this case, Button), and give the parameters.

from tkinter import *

window = Tk()

window.mainloop()

# widgets go here....

# remember that for creating the Widgets, first parameter it takes is, where to create, i.e. the Window you created.

label = Label(window, text='This is a Label')

button = Button(window, text='click me')

# We will now pack it.

label.pack()
